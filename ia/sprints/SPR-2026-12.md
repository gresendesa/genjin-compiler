### `Troca_Ferramenta(home_gastos: &Text, home_novos: &Text, home_voltar: &Text)`

Aparece 1x no `lenhador.gnj` (caso `FERRAMENTA_GASTA`, ~25 linhas), mas é o padrão central
do LenhadorNEO — criado como proc-bloco para reutilização futura com outras ferramentas.

```gnj
Troca_Ferramenta(home_gastos: &Text, home_novos: &Text, home_voltar: &Text) {
    exec Definir_Home(home=&home_voltar) as "Troca:Definir_Home" {
        case SUCESSO:
            exec Teleportar(home=&home_gastos) as "Troca:TP_Gastos" {
                case SUCESSO:
                    exec Guardar_Itens_MKB(gaveta="gaveta_mkb_guardar", itens="['diamond_axe']", quantidade=1, quantidade_movidos_var=&quantidade_movidos) as "Troca:Guardar" {
                        case SUCESSO:
                            exec Teleportar(home=&home_novos) as "Troca:TP_Novos" {
                                case SUCESSO:
                                    exec Pegar_Itens_MKB(gaveta="gaveta_mkb_pegar", itens="['diamond_axe']", quantidade=1, quantidade_movidos_var=&quantidade_movidos) as "Troca:Pegar" {
                                        case SUCESSO:
                                            exec Teleportar(home=&home_voltar) as "Troca:Voltar" {
                                                pass SUCESSO, NÃO_EXISTE, CANCELADO
                                            } while(DEMORA)
                                    } while(DEMORA)
                                pass NÃO_EXISTE, CANCELADO
                            } while(DEMORA)
                    } while(DEMORA)
                pass NÃO_EXISTE, CANCELADO
            } while(DEMORA)
        pass SUCESSO, NÃO_EXISTE, REINICIAR_CICLO, LIMITE_ALCANCADO, PROIBIDO
    } while(DEMORA, CANCELADO)
}
```

Codes inferidos: `SUCESSO`, `NÃO_EXISTE`, `REINICIAR_CICLO`, `LIMITE_ALCANCADO`, `PROIBIDO`
